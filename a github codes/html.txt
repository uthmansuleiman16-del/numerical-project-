// =============================
// ROOT METHODS
// =============================
function solveRoot(){
let fx=f(document.getElementById("fx").value);
let df=f(document.getElementById("dfx").value);
let x0=parseFloat(document.getElementById("x0").value);
let x1=parseFloat(document.getElementById("x1").value);
let method=document.getElementById("method").value;

let root;
if(method=="bisection") root=(x0+x1)/2;
if(method=="newton") root=x0 - fx(x0)/df(x0);

print([root]);
plot(fx);
}

// =============================

function plot(fx){
let c=document.getElementById("graph");
let ctx=c.getContext("2d");
ctx.clearRect(0,0,c.width,c.height);
ctx.beginPath();
for(let x=-10;x<=10;x+=0.1){
let y=fx(x);
let px=(x+10)*25;
let py=150-y*10;
if(x===-10) ctx.moveTo(px,py);
else ctx.lineTo(px,py);
}
ctx.stroke();
}

// =============================
// MATRIX PARSER
// =============================
function parseM(t){ return t.split("\n").map(r=>r.split(",").map(Number)); }

// =============================
// GAUSS-JORDAN
// =============================
function gaussJordan(A,b){
let n=A.length;
for(let i=0;i<n;i++){
let diag=A[i][i];
for(let j=0;j<n;j++) A[i][j]/=diag;
b[i]/=diag;
for(let k=0;k<n;k++){
if(k!==i){
let factor=A[k][i];
for(let j=0;j<n;j++) A[k][j]-=factor*A[i][j];
b[k]-=factor*b[i];
}
}
}
return b;
}

// =============================
// LU DECOMPOSITION
// =============================
function lu(A,b){
let n=A.length;
let L=Array.from({length:n},()=>Array(n).fill(0));
let U=Array.from({length:n},()=>Array(n).fill(0));

for(let i=0;i<n;i++){
L[i][i]=1;
for(let j=i;j<n;j++){
U[i][j]=A[i][j];
for(let k=0;k<i;k++) U[i][j]-=L[i][k]*U[k][j];
}
for(let j=i+1;j<n;j++){
L[j][i]=A[j][i];
for(let k=0;k<i;k++) L[j][i]-=L[j][k]*U[k][i];
L[j][i]/=U[i][i];
}
}

let y=new Array(n);
for(let i=0;i<n;i++){
y[i]=b[i];
for(let k=0;k<i;k++) y[i]-=L[i][k]*y[k];
}

let x=new Array(n);
for(let i=n-1;i>=0;i--){
x[i]=y[i];
for(let k=i+1;k<n;k++) x[i]-=U[i][k]*x[k];
x[i]/=U[i][i];
}

return x;
}

// =============================
// LINEAR SOLVER
// =============================
function solveLinear(){
let A=parseM(document.getElementById("A").value);
let b=document.getElementById("B").value.split(",").map(Number);
let method=document.getElementById("linearMethod").value;

let res;
if(method=="jordan") res=gaussJordan(A,b);
if(method=="lu") res=lu(A,b);

print(res);
}

// =============================
// LAGRANGE INTERPOLATION (CH3)
// Formula builds polynomial using all points
// =============================
function lagrange(xVals,yVals,x){
let n=xVals.length;
let result=0;

for(let i=0;i<n;i++){
let term=yVals[i];

for(let j=0;j<n;j++){
if(i!==j){
term *= (x - xVals[j])/(xVals[i]-xVals[j]);
}
}

result+=term;
}

return result;
}

// =============================
// INTERPOLATION SOLVER
// =============================
function solveInterp(){
let xVals=document.getElementById("xVals").value.split(",").map(Number);
let yVals=document.getElementById("yVals").value.split(",").map(Number);
let x=parseFloat(document.getElementById("xTarget").value);

let result=lagrange(xVals,yVals,x);

print([result]);
}
